document.querySelectorAll('.tab-btn').forEach(btn => { btn.addEventListener('click', () => { document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active')); btn.classList.add('active'); const country = btn.getAttribute('data-country'); document.querySelectorAll('.country-testimonials').forEach(c => { c.classList.remove('active'); }); const container = document.getElementById(country); container.classList.add('active'); // Fade-in testimonials container.querySelectorAll('.testimonial-item').forEach((item, index) => { item.style.animation = `fadeUp 0.5s forwards ${index * 0.2}s`; }); loadTestimonials(country); }); }); function loadTestimonials(country) { const container = document.getElementById(country); container.innerHTML = ''; testimonials[country].forEach(testimonial => { const stars = '★'.repeat(testimonial.rating) + '☆'.repeat(5 - testimonial.rating); container.innerHTML += `

"${testimonial.text}"

${testimonial.name} ${testimonial.location}
${stars}
`; }); } loadTestimonials('all');
